# Plunder Alliance Digsites
estate_reliquary_plunder_alliance_digsites = {
	icon = privilege_jack_of_trades

    # Costs
    max_absolutism = -5
	land_share = 0
	loyalty = 0.05
	influence = 0.05
    
    cooldown_years = 10
    
    is_valid = {
        any_known_country = {
            has_country_modifier = member_of_alliance
            has_estate = estate_explorers_league
        }
    }
    can_select = {
        any_known_country = {
            has_country_modifier = member_of_alliance
            has_estate = estate_explorers_league
        }
    }
	benefits = {
        
	}
    on_granted = {
        random_known_country = {
            limit = {
                has_country_modifier = member_of_alliance
                has_estate = estate_explorers_league
            }
            add_opinion = {
                who = ROOT
                modifier = plundered_digsite
                years = 5
            }
            
            if = {
                limit = {
                    total_development = 100
                }
                
                ROOT = {
                    add_treasury = 200
                }
            }
            else_if = {
                limit = {
                    total_development = 250
                }
                
                ROOT = {
                    add_treasury = 500
                }
            }
            else_if = {
                limit = {
                    total_development = 500
                }
                
                ROOT = {
                    add_treasury = 1000
                }
            }
            else = {
                ROOT = {
                    add_treasury = 100
                }
            }
        }
	}
    
	ai_will_do = {
		factor = 10
	}
}
