# INSTRUCTIONS:
#
# condition				A diplomatic action can have any number of condition blocks, each with its own
#						tooltip, potential and allow section
#
# 	tooltip					Sets a custom text string similar to the hardcoded limits
# 							If no tooltip is scripted, the tooltip for the actual trigger will be shown
#							Note that the custom tooltip is only shown if the allow trigger is NOT met
#
# 	potential				Determines if the trigger is applicable or not
# 	allow					Determines if the action is valid or not
#
# effect				A diplomatic action can only have one effect block

# ROOT					actor
# FROM					target

royal_marriage = {
    effect = {
        if = { 
            limit = { has_global_flag = enable_diplomacy_logging }
            log = "DEBUG - [Root.GetName], [From.GetName] - royal_marriage"
        }
    }
    
	condition = {
		tooltip = DIP_ENFORCED_ISOLATION_THEM
		potential = {
            has_global_flag = enabled_isolation_mechanics
            FROM = {
                has_country_modifier = enforced_isolation
            }
		}
		allow = {
			always = no
		}
	}
    condition = {
		tooltip = DIP_ENFORCED_ISOLATION_US
		potential = {
            has_global_flag = enabled_isolation_mechanics
            has_country_modifier = enforced_isolation
		}
		allow = {
			always = no
		}
	}
    
    condition = {
		tooltip = MARWAR
		potential = {
		}
		allow = {
			NOT = { war_with = FROM }
		}
	}
}

declarewar = {
    effect = {
        if = { 
            limit = { has_global_flag = enable_diplomacy_logging }
            log = "DEBUG - [Root.GetName], [From.GetName] - declarewar"
        }
    
        # Aggressive
        if = {
            limit = {
                has_reform = dragonflight_brood_reform
            }
            
            random_list = {
                20 = { change_statists_vs_monarchists = -0.05 }
                20 = { change_statists_vs_monarchists = -0.04 }

                20 = { change_statists_vs_monarchists = -0.02 }

        tooltip = PREVENT_WAR_DIP_TT

            FROM = {
                has_country_modifier = chi_prevent_war
            }
		}
        allow = {
            always = no
        }

    condition = {
		tooltip = DIP_ENFORCED_ISOLATION_US
		potential = {
            has_global_flag = enabled_isolation_mechanics
            has_country_modifier = enforced_isolation
		}
		allow = {
			always = no
		}

allianceaction = {  
    effect = {
        if = { 
            limit = { has_global_flag = enable_diplomacy_logging }
            log = "DEBUG - [Root.GetName], [From.GetName] - allianceaction"
        }
    
        # Diplomatic
        if = {
            limit = {
                has_reform = dragonflight_brood_reform
            }
            
            random_list = {
                20 = { change_statists_vs_monarchists = 0.05 }
                20 = { change_statists_vs_monarchists = 0.04 }
                20 = { change_statists_vs_monarchists = 0.03 }
                20 = { change_statists_vs_monarchists = 0.02 }
                20 = { change_statists_vs_monarchists = 0.01 }
            }
        }
    }
    
    condition = {
        tooltip = DIP_INCOMP_WORGEN_FORSAKEN
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
			OR = {
				AND = {
                    culture_group = culture_group_worgen
                    FROM = {
                        culture_group = culture_group_undead
                    }
                }
                AND = {
                    culture_group = culture_group_undead
                    FROM = {
                        culture_group = culture_group_worgen
                    }
                }
			}
		}
        allow = {
            always = no
        }
	}
    
    condition = {
        tooltip = DIP_INCOMP_EVIL
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
			OR = {
				AND = {
					OR = {
						religion = cthun
						religion = nzoth
						religion = yogg_saron
						religion = yshaarj
						religion = twilight_cult
						religion = nightmare
					}
					FROM = {
						NOT = { religion_group = shadow_group }
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					religion_group = disorder_group
					FROM = {
						NOT = { religion_group = disorder_group }
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					NOT = { religion_group = shadow_group }
					FROM = {
						OR = {
							religion = cthun
							religion = nzoth
							religion = yogg_saron
							religion = yshaarj
							religion = twilight_cult
							religion = nightmare
						}
					}
				}
				AND = {
					NOT = { religion_group = disorder_group }
					FROM = {
						religion_group = disorder_group
					}
				}
				AND = {
					NOT = { religion_group = death_group }
					FROM = {
						religion_group = death_group
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    condition = {
        tooltip = DIP_INCOMP_SCOURGE
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
			OR = {
				AND = {
					religion = cult_of_the_damned
					FROM = {
						NOT = { religion_group = shadow_group }
                        NOT = { religion_group = death_group }
					}
				}
				AND = {
                    NOT = { religion_group = shadow_group }
                    NOT = { religion_group = death_group }
                    
					FROM = {
						religion = cult_of_the_damned
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    condition = {
        tooltip = DIP_INCOMP_UNCIVILIZED
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
            
			OR = {
				AND = {
					is_uncivilized = yes
					FROM = {
						NOT = { is_uncivilized = yes }
					}
				}
				AND = {
					NOT = { is_uncivilized = yes }
					NOT = { is_savage = yes }
					NOT = { is_neighbor_of = FROM }
					NOT = { religion_group = shadow_group }
					NOT = { religion_group = death_group }
					NOT = { religion_group = disorder_group }
					FROM = {
						is_uncivilized = yes
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    condition = {
        tooltip = DIP_INCOMP_FACTION
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
            
			OR = {
				AND = {
					OR = {
						has_country_modifier = head_of_alliance
                        has_country_modifier = member_of_alliance
					}
					FROM = {
						OR = {
							has_country_modifier = head_of_horde
                            has_country_modifier = member_of_horde
                            has_country_modifier = head_of_old_horde
                            has_country_modifier = member_of_old_horde
						}
					}
				}
				AND = {
					OR = {
                        has_country_modifier = head_of_horde
                        has_country_modifier = member_of_horde
                        has_country_modifier = head_of_old_horde
                        has_country_modifier = member_of_old_horde
                    }
					FROM = {
						OR = {
                            has_country_modifier = head_of_alliance
                            has_country_modifier = member_of_alliance
                        }
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    condition = {
        tooltip = DIP_INCOMP_HISTORICAL
        
        potential = {
            has_global_flag = enabled_lore_diplomacy_mechanics
            
			OR = {
				# Humans - Orcs
				AND = {
					culture_group = culture_group_human
					FROM = { culture_group = culture_group_orc }
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
				AND = {
					culture_group = culture_group_orc
					FROM = { culture_group = culture_group_human }
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
				# Elves - Trolls
				AND = {
					OR = {
						culture_group = culture_group_thalassian
						culture_group = culture_group_darnassian
					}
					FROM = { 
						OR = {
							primary_culture = culture_forest_troll 
							primary_culture = culture_jungle_troll 
							primary_culture = culture_dark_troll 
							primary_culture = culture_frost_troll
							primary_culture = culture_sand_troll
							primary_culture = culture_zandalari_troll
						}
					}
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
				AND = {
					OR = {
						primary_culture = culture_forest_troll 
						primary_culture = culture_jungle_troll 
						primary_culture = culture_dark_troll 
						primary_culture = culture_frost_troll
						primary_culture = culture_sand_troll
						primary_culture = culture_zandalari_troll
					}
					FROM = { 
						OR = {
							culture_group = culture_group_thalassian
							culture_group = culture_group_darnassian
						}
					}
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
				#Worgen - Undeads
				AND = {
					culture_group = culture_group_worgen
					FROM = { 
						culture_group = culture_group_undead
					}
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
				AND = {
					culture_group = culture_group_undead
					FROM = { 
						culture_group = culture_group_worgen
					}
					NOT = {
						any_known_country = {
							alliance_with = ROOT
							alliance_with = FROM
						}
					}
				}
			}
		}
        allow = {
            always = no
        }
	}
    
    condition = {
		tooltip = DIP_ENFORCED_ISOLATION_THEM
		potential = {
            has_global_flag = enabled_isolation_mechanics
            FROM = {
                has_country_modifier = enforced_isolation
            }
		}
		allow = {
			always = no
		}
	}
    condition = {
		tooltip = DIP_ENFORCED_ISOLATION_US
		potential = {
            has_global_flag = enabled_isolation_mechanics
            has_country_modifier = enforced_isolation
		}
		allow = {
			always = no
		}
	}
}
