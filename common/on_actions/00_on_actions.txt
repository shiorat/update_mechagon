# country
on_startup = {
	if = { 
		limit = { has_global_flag = enable_onaction_logging }
		log = "DEBUG - [Root.GetName]: on_startup "
	}
    
    emperor = {
		# Audax Validator "." Ignore_NEXT
		save_global_event_target_as = Emperor
	}
    
    # Default "Old Horde Leader" event target to Shadowmoon
    A34 = {
		# Audax Validator "." Ignore_NEXT
        save_global_event_target_as = old_horde_leader
    }
    
	if = {
		limit = {
			is_emperor_of_china = yes
		}
		# Audax Validator "." Ignore_NEXT
		save_global_event_target_as = celestial_emperor
	}
    
    # Set the first player nation as the Host Player, which will receive the single-fire player only events in multiplayer.
    if = {
        limit = {
            ai = no
            NOT = { has_global_flag = set_host_player }
        }
        
        set_global_flag = set_host_player
        set_country_flag = host_player
    }
    
    unlock_initial_cults = yes
    
    events = {
        wwu_setup.1         # Global
        wwu_setup.2         # Country
        wwu_loa.1           # Troll - Starting Loa Influence
        wwu_zandalar.100    # Zandalar Setup
        wwu_game_configuration.100 # Default Configuration
        wwu_doctrine.100    # Default
	}
}

on_war_won = {
	if = { 
		limit = { has_global_flag = enable_onaction_logging }
		log = "DEBUG - [Root.GetName], [From.GetName]: on_war_won "
	}
    
	ROOT = { 
		country_event = { id = tributary_events.32 }
		change_variable = {
			which = war_won
			value = 1
		}
        
        # Doctrine Notification Reset
        country_event = { id = wwu_doctrine.99 days = 3 }
	}
    
    FROM = {
        # Doctrine Notification Reset
        country_event = { id = wwu_doctrine.99 days = 3 }
    }
    
    if = {
        limit = {
            tag = A66 # Gurubashi
            FROM = {
                tag = STW
            }
        }
        set_country_flag = won_war_against_stormwind
    }
    
    # War ends - these should only apply if the war end causes the winner to end the war
    if = {
        limit = {
            is_at_war = no
        }
        
        if = {
            limit = {
                is_nomad = yes
            }
            
            add_horde_unity = 20
        }
        
        if = {
            limit = {
                has_estate_privilege = estate_tribals_spoils_of_war
            }
            add_estate_loyalty = {
                estate = estate_tribals
                loyalty = 20
            }
        }
        
        # Reset Invasion flag
        clr_country_flag = invasion_spawn_done
        clr_country_flag = first_core_of_invasion
    }
    
	events = {
        wwu_dragon_aspects.1
		on_action_special.2
        wwu_elemental_attunement.1 # Triumph of the Elements 
        wwu_nightfallen.101 # Arcane Siphoning
        wwu_artifact_unlocks.3 # Blades of the Fallen Prince
        wwu_mantid.4 # Lessons from the Cycle
	}
}

# root = winning country, from = loser country
on_battle_won_country = {
	if = { 
		limit = { has_global_flag = enable_onaction_logging }
		log = "DEBUG - [Root.GetName], [From.GetName]: on_battle_won_country "
	}

    # Faction: Alliance Fervor
    if = {
        limit = {
            has_country_modifier = member_of_alliance
            FROM = {
                OR = {
                    has_country_modifier = member_of_horde
                    has_country_modifier = member_of_old_horde
                }
            }
        }
        
        add_country_modifier = {
            name = factional_fervor
            duration = 365
        }
    }

    # Faction: Horde Fervor
    if = {
        limit = {
            OR = {
                has_country_modifier = member_of_horde
                has_country_modifier = member_of_old_horde
            }
            FROM = {
                has_country_modifier = member_of_alliance
            }
        }
        
        add_country_modifier = {
            name = factional_fervor
            duration = 365
        }
    }
    
    # Government: Nomad
    if = {
        limit = {
            is_nomad = yes
            FROM = {
                NOT = { tag = REB }
            }
        }
        
        add_horde_unity = 5
    }
    
    # Reform: War Propaganda
    if = {
        limit = {
            has_estate_privilege = estate_peasants_war_propaganda
        }
        
        add_prestige = 5
    }

    # Reform: Idealism
    if = {
        limit = {
            has_reform = wwu_idealism_glory
        }
        add_power_projection = {
            type = mission_rewards_power_projection
            amount = 5
        }
    }

	if = {
		limit = {
			FROM = { tag = REB }
		}
		change_variable = {
			which = rebels_suppressed
			value = 1
		}
	}
    
    # Raise the Dead
    country_event = { id = wwu_scourge_estate.3 }
    country_event = { id = wwu_scourge_estate.4 }
    
    random_events = {
        2000 = 0
        100 = wwu_mantid.7 # Mantid Trophies
        100 = wwu_yaungol.4 # Yaungol Pride
    }
}

            culture_group = culture_group_troll

            FROM = {
                custom_trigger_tooltip = {
                    tooltip = IS_PRIMARY_ARTIFACT_stromkar
                    has_country_flag = primary_artifact_stromkar
                }
            }
        }
        
        # Loser
        country_event = { id = wwu_archaeology.511 }

        FROM = {
            country_event = { id = wwu_archaeology.510 }
        }
    }
    
    # Ideagroup: Trial by Combat

        limit = {
            # Loser has a general

                value = 1
                type = general
                include_monarch = no
                include_heir = no
            }
            
            # Winner has Trial by Combat

                has_idea = wwu_law_ideas_2

            25 = {
                # Loser event

                FROM = {

    if = {
        limit = {
            has_estate_privilege = estate_peasants_war_propaganda

        }
    }

    if = {
        limit = {
            has_government_attribute = enables_enslave_captives_event
        }
        random = {
            chance = 10
            
            country_event = { id = wwu_tribal.2 }
        }
    }
    
    # Integrate Captives 
    if = {
        limit = {
            has_government_attribute = enables_integrate_captives_event
        }
        random = {
            chance = 10
            
            country_event = { id = wwu_tribal.3 }
        }
    }
    
    # Execute Captives 
    if = {
        limit = {
            has_government_attribute = enables_execute_captives_event
        }
        random = {
            chance = 10
            
            country_event = { id = wwu_tribal.4 }
        }
    }
    
    # Conscript Captives 
    if = {
        limit = {
            has_government_attribute = enables_conscript_captives_event
        }
        random = {
            chance = 10
            
            country_event = { id = wwu_tribal.5 }
        }
    }
    
    # Invasion - Add core to first province sieged
    if = {
        limit = {
            NOT = { has_country_flag = first_core_of_invasion }

        which = ruler_magic_training

		wwu_loa.1 # Influence of X

        wwu_shamanism.1 # Passing of Great Leader
	}
    
    random_events = {
        100 = wwu_government.12 # Puppet Ruler
        
        200 = 0
    }
	
}

	}
    on_annexed_estate_privileges_effect = yes
    
    country_event = { id = wwu_dragon_aspects.30 } # Dismantled Dragon Aspects
    country_event = { id = wwu_dragon_aspects.31 } # Claim the Dragon Aspects

	change_variable = {
		which = annexed_nation
		value = 1
	}
    
    # If Head of faction is annexed, re-assign leadership
    if = {
        limit = {
            FROM = {
                has_country_modifier = head_of_alliance
            }
        }
        country_event = { id = wwu_faction_system.30 }
    }

            FROM = {
                has_country_modifier = head_of_old_horde
            }
        }
        country_event = { id = wwu_faction_system.31 }
    }

        limit = {
            FROM = {
                has_country_modifier = head_of_horde
            }
        }
        country_event = { id = wwu_faction_system.32 }
    }

        limit = {
            FROM = {

		}
	}

        limit = {
            has_province_modifier = religious_persecution
        }
        remove_province_modifier = religious_persecution
    }
	
	if = {
		limit = {
			religion = fel
			trade_goods = arcane
		}
		change_trade_goods = fel
	}
	
	if = {
		limit = {
			religion = cult_of_the_damned
			trade_goods = arcane
		}
		change_trade_goods = fel
	}
	
	if = {
		limit = {
			NOT = { religion = fel }
			NOT = { religion = cult_of_the_damned }
			trade_goods = fel
		}
		change_trade_goods = arcane
	}
    
    on_province_religion_converted_estate_privileges_effect = yes
}
