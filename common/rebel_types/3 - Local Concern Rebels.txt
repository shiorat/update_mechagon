#-------------------------------------------------
# Bandits
#-------------------------------------------------
lc_bandit_rebels = {

	color = { 50 85 210 }

	area = nation 					# Stays within the nation
	government = any	
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	gfx_type = culture_province
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	unit_transfer = no
	will_relocate = no
	
	infantry = 0.8
	cavalry = 0.1
	artillery = 0.1
	
	morale = 1.5
	
	# Possible handle actions
	handle_action_negotiate = no
	handle_action_stability = no
	handle_action_build_core = no
	handle_action_send_missionary = no
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 2
        
        # Not Present
		modifier = {
			factor = 0
			has_no_local_concern = {
                TYPE = bandit
            }
		}
        
        # Level 1
		modifier = {
			factor = 2
			has_province_modifier = lc_level_bandit_1
		}
        
        # Level 2
		modifier = {
			factor = 3
			has_province_modifier = lc_level_bandit_2
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 10
			is_capital = yes
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		always = yes
 	}
	siege_won_effect = {
		hidden_effect = { 
            add_devastation = 5 
        }
        
		add_permanent_province_modifier = {
			name = devasted_province
			duration = 1095
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		always = no
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = "lc_bandit_rebels_demand"
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		add_years_of_income = -5
		add_stability = -1
	}
}
