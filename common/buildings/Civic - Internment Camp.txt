# AI will consider buildings it does not have the tech for at 20% of scripted AI weight (in case it wants to save slots)

wow_internment_camp_1 = {
	cost = 100
	time = 12
	
	modifier = {
		local_unrest = -1
        local_culture_conversion_cost = -0.25
	}

    on_built = {
        if = { 
			limit = { has_global_flag = enable_building_logging }
			log = "DEBUG - [Root.GetName] - on_built - wow_internment_camp_1 "
        }
        
        add_development_from_building = {
			building = wow_internment_camp_1
			type = manpower
			value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = wow_internment_camp_1
			type = manpower
			value = 1
		}
	}
	on_obsolete = {
	}
    
	ai_will_do = {
		factor = 1.0
        
        # Only let AI build these on Orc culture provinces
        modifier = {
            factor = 0
            
            NOT = { culture = culture_orc }
        }
	}
}
wow_internment_camp_2 = {
	cost = 300
	time = 24
	make_obsolete = wow_internment_camp_1
    
	modifier = {
		local_unrest = -2
        local_culture_conversion_cost = -0.5
	}

    on_built = {
        if = { 
			limit = { has_global_flag = enable_building_logging }
			log = "DEBUG - [Root.GetName] - on_built - wow_internment_camp_2 "
		}
        
		add_development_from_upgrading_or_building = {
			building = wow_internment_camp_2
			obsolete_building = wow_internment_camp_1
			type = manpower
			value = 2
			upgrade_value = 1
		}
	}
	on_destroyed = {
		remove_development_from_building = {
			building = wow_internment_camp_2

			value = 2
		}
	}
	on_obsolete = {
	}
    
	ai_will_do = {
		factor = 1.0
        
        # Only let AI build these on Orc culture provinces
        modifier = {
            factor = 0
            
            NOT = { culture = culture_orc }
        }
	}
}
