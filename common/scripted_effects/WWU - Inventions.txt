increment_research = {
    custom_tooltip = INCREMENT_RESEARCH_TT
    
    hidden_effect = {
        change_variable = {
            which = research_progress
            value = 1
        }
        
        if = {
            limit = { 
                has_country_flag = research_progress_boost
            }
            
            change_variable = {
                which = research_progress
                value = 1
            }
        }
        
        if = {
            limit = {
                check_variable = {
                    which = research_progress
                    value = 5
                }
            }
            set_variable = {
                which = research_progress
                value = 5
            }
        }
    }
}

}

decrement_research = {
    custom_tooltip = DECREMENT_RESEARCH_TT
    
    hidden_effect = {
        subtract_variable = {
            which = research_progress
            value = 1
        }
        
        if = {
            limit = {
                NOT = { 
                    check_variable = {
                        which = research_progress
                        value = 0
                    }
                }
            }
            set_variable = {
                which = research_progress
                value = 0
            }
        }
    }
}
