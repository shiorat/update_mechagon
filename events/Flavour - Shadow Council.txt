namespace = wwu_shadow_council

# The Shadow Council
country_event = {
	id = wwu_shadow_council.1
	title = wwu_shadow_council.1.title
	desc = wwu_shadow_council.1.desc
	picture = GULDAN_eventPicture
    
	major = yes
    fire_only_once = yes
    
	trigger = {
        tag = B88
	}
	
	mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_shadow_council.1.option.a
		
        define_ruler = { 
            name = "Gul'dan" 
            dynasty = "Shadowmoon" 
            adm = 5
            dip = 5
            mil = 3
            age = 80
        }
        set_ruler_flag = is_immortal

        setup_spellcaster_history = {
            APTITUDE = 4
            FINESSE = 4
            CONCENTRATION = 4
            MANA = 5
            fel = yes
        }
        
        add_country_modifier = {
            name = the_shadow_council
            duration = -1
        }	
        
        if = {
            limit = {
                any_country = {
                    has_country_modifier = head_of_old_horde
                }
            }
            
            set_member_of_faction = {
                faction = old_horde
            }
        }
	}
}
