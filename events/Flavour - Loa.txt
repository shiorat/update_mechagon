	trigger = {
        has_estate = estate_loa

		has_any_loa = yes

			NOT = { has_province_modifier = shrine_to_loa } 
		}

	}
    
    immediate = {
        hidden_effect = {
            add_country_modifier = {
                name = loa_event_cooldown
                duration = 3650
                hidden = yes
            }
        }
    }
	
	mean_time_to_happen = {
		months = 120
	}
		
	option = {
		name = wwu_loa.10.option.a
		
		add_treasury = -100
        
		random_owned_province = {
			limit = { 
                NOT = { has_province_modifier = shrine_to_loa }   
            }
            
			add_permanent_province_modifier = {
				name = shrine_to_loa
				duration = 18250
			}
		}
        
		random_list = { 
			20 = { increase_loa_influence = yes }
            80 = { custom_tooltip = NOTHING_HAPPENS }
		}
	}
	option = {
		name = wwu_loa.10.option.b
		
		add_adm_power = 50
	}
}

# Devastation of X
country_event = {
	id = wwu_loa.15
	title = wwu_loa.15.title
	desc = wwu_loa.15.desc
	picture = DECIMATE_eventPicture
    
	trigger = { 
        has_estate = estate_loa
        
        NOT = { has_country_modifier = loa_event_cooldown }
        
		has_any_loa = yes
        
        treasury = 100
        mil_power = 50
        
        any_owned_province = {
            any_neighbor_province = {
                owned_by = ROOT
            }
        }
	}
	
	immediate = {
        hidden_effect = {
            add_country_modifier = {
                name = loa_event_cooldown
                duration = 3650
                hidden = yes
            }
            
            random_owned_province = {
                save_event_target_as = own_province
                
                random_neighbor_province = {
                    save_event_target_as = neighbor_province
                }
            }
        }
    }
	
	mean_time_to_happen = {
		months = 360
	}
		
	option = {
		name = wwu_loa.15.option.a
		
		add_treasury = -100
        
		event_target:own_province = {
			add_devastation = 25
		}
        
		random_list = { 
			20 = { increase_loa_influence = yes }
            80 = { custom_tooltip = NOTHING_HAPPENS }
		}
	} 
	option = {
		name = wwu_loa.15.option.b
		
		add_mil_power = -50
        
		event_target:own_province = {
			add_devastation = 25
		}
        
		random_list = { 
			20 = { decrease_loa_influence = yes }
            80 = { custom_tooltip = NOTHING_HAPPENS }
		}
	}
	option = {
		name = wwu_loa.15.option.c
		
		event_target:neighbor_province = {
			add_devastation = 50
		}
	}
}
