# Intimidation
country_event = {
	id = wwu_diplomatic_actions.3
	title = wwu_diplomatic_actions.3.title
    desc = wwu_diplomatic_actions.3.desc
	picture = ABERRATION_eventPicture

    is_triggered_only = yes
    
	mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            export_to_variable = {
                which = our_MIL
                who = ROOT
                value = trigger_value:MIL
            }
            export_to_variable = {
                which = enemy_MIL
				# Audax Validator "." Ignore_NEXT
                who = event_target:intimidator
                value = trigger_value:MIL
            }
            
            set_variable = {
                which = diff_MIL
                value = 0
            }
            change_variable = {
                which = diff_MIL
                which = our_MIL
            }
            subtract_variable = {
                which = diff_MIL
                which = enemy_MIL
            }
        }
    }

	option = {
		name = wwu_diplomatic_actions.3.option.a
		
        # Good (+2 or higher)
        if = {
            limit = {
                check_variable = {
                    which = diff_MIL
                    value = 2
                }
            }
            
            # Success Roll
            random_list = {
                # Intimidated
                50 = {
					# Audax Validator "." Ignore_NEXT
                    event_target:intimidator = {
                        add_truce_with = ROOT
                    }
					# Audax Validator "." Ignore_NEXT
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.4 }
                    }
                }
                # Unfazed
                50 = {
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.5 }
                    }
                }
            }
        }
        # Neutral (1 to -1)
        else_if = {
            limit = {
                NOT = {
                    check_variable = {
                        which = diff_MIL
                        value = 2
                    }
                }
                check_variable = {
                    which = diff_MIL
                    value = -1
                }
            }
            
            # Success Roll
            random_list = {
                # Intimidated
                75 = {
                    event_target:intimidator = {
                        add_truce_with = ROOT
                    }
                
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.4 }
                    }
                }
                # Unfazed
                25 = {
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.5 }
                    }
                }
            }
        }
        # Bad
        else_if = {
            limit = {
                NOT = {
                    check_variable = {
                        which = diff_MIL
                        value = -1
                    }
                }
            }
            
            # Success Roll
            random_list = {
                # Intimidated
                90 = {
                    event_target:intimidator = {
                        add_truce_with = ROOT
                    }
                
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.4 }
                    }
                }
                # Unfazed
                10 = {
                    event_target:intimidator = {
                        country_event = { id = wwu_diplomatic_actions.5 }
                    }
                }
            }
        }
        
        ai_chance = {
            factor = 100
        }
	}
    
    option = {
		name = wwu_diplomatic_actions.3.option.b
        
        event_target:intimidator = {
            country_event = { id = wwu_diplomatic_actions.5 }
        }
        
        add_dip_power = -250
        
        ai_chance = {
            factor = 0
        }
	}
}

# A Bribe for Peace
country_event = {
	id = wwu_diplomatic_actions.21
	title = wwu_diplomatic_actions.21.title
    desc = wwu_diplomatic_actions.21.desc
	picture = SACK_OF_COINS_eventPicture

    is_triggered_only = yes

	mean_time_to_happen = {
		days = 1
	}

    # Accept
	option = {
		name = wwu_diplomatic_actions.21.option.a
        
        if = {
            limit = {
                has_country_flag = small_bribe
            }
            add_treasury = 100
        }
        if = {
            limit = {
                has_country_flag = medium_bribe
            }
            add_treasury = 300
        }
        if = {
            limit = {
                has_country_flag = large_bribe
            }
            add_treasury = 500
        }
        if = {
            limit = {
                has_country_flag = huge_bribe
            }
            add_treasury = 1000
        }
        
        event_target:briber = {
            country_event = { id = wwu_diplomatic_actions.22 days = 1 }
        }
        
        ai_chance = {
            factor = 80
        }
	}
    # Reject
    option = {
		name = wwu_diplomatic_actions.21.option.b
        
        # Some of the bribe is 'lost' if rejected
        if = {
            limit = {
                has_country_flag = small_bribe
            }
            add_treasury = 75
        }
        if = {
            limit = {
                has_country_flag = medium_bribe
            }
            add_treasury = 250
        }
        if = {
            limit = {
                has_country_flag = large_bribe
            }
            add_treasury = 400
        }
        if = {
            limit = {
                has_country_flag = huge_bribe
            }
            add_treasury = 800
        }
		# Audax Validator "." Ignore_NEXT
        event_target:briber = {
            country_event = { id = wwu_diplomatic_actions.23 days = 1 }
        }
        
        ai_chance = {
            factor = 20
            
            # Boost if bribe is too small
            modifier = {
                factor = 2
                
                total_development = 300
                has_country_flag = small_bribe
            }
            
            # Boost if bribe is too small
            modifier = {
                factor = 2
                
                total_development = 500
                OR = {
                    has_country_flag = small_bribe
                    has_country_flag = medium_bribe
                }
            }
            
            # Reduce if bribe is huge and nation isn't massive
            modifier = {
                factor = 0.5
                
                NOT = { total_development = 1000 }
                has_country_flag = huge_bribe
            }
            
            # Reduce if bribe is huge and nation isn't massive
            modifier = {
                factor = 0.5
                
                NOT = { total_development = 500 }
                has_country_flag = huge_bribe
            }
            
            # Reduce if bribe is huge and nation isn't massive
            modifier = {
                factor = 0.5
                
                NOT = { total_development = 250 }
                OR = {
                    has_country_flag = large_bribe
                    has_country_flag = huge_bribe
                }
            }
        }
	}
    
    after = {
        hidden_effect = {
            clr_country_flag = small_bribe
            clr_country_flag = medium_bribe
            clr_country_flag = large_bribe
            clr_country_flag = huge_bribe
        }
    }
}
