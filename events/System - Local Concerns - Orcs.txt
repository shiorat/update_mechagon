# Local Migration
province_event = {
    id = wwu_lc_orcs.10
    title = wwu_lc_orcs.10.title
    desc = wwu_lc_orcs.10.desc
    picture = FROSTWOLF_WARLORD_eventPicture
    goto = ROOT
    
    is_triggered_only = yes
    
    trigger = {
        OR = {
            has_local_concern_level_1 = { TYPE = orc }
            has_local_concern_level_2 = { TYPE = orc }
        }

        any_neighbor_province = {
            is_city = yes
            
            has_no_local_concern = {
                TYPE = orc
            }
            
            NOT = {
                owner = {
                    is_orc_type_nation = yes
                }
            }

            NOT = {
                has_banned_terrain = yes
            }
        }
    }
    
    immediate = {
        hidden_effect = {
            random_neighbor_province = {
                limit = {
                    is_city = yes

                    has_no_local_concern = {
                        TYPE = orc
                    }
                    
                    NOT = {
                        owner = {
                            is_orc_type_nation = yes
                        }
                    }
                    
                    NOT = {
                        has_banned_terrain = yes
                    }
                }
                
                save_event_target_as = lc_orc_target
            }
        }
    }

    # Same Country
    option = {  
        name = wwu_lc_orcs.10.option.a
        goto = lc_orc_target
        
        ai_chance = { factor = 100 }
        
        trigger = {
            event_target:lc_orc_target = {
                owned_by = ROOT
            }
        }

        # Remove orcs
        tooltip = {
            remove_local_concern = {
                TYPE = orc
            }
        }

        # Add orcs to the new province based on the previous province tier
        event_target:lc_orc_target = {
            if = {
                limit = {
                    owned_by = ROOT
                }

                upgrade_local_concern = {
                    TYPE = orc
                }
                
                # Remove orcs from the previous province
                PREV = {
                    remove_local_concern = {
                        TYPE = orc
                    }
                }
            }
        }
    }

    # Different Country
    option = {  
        name = wwu_lc_orcs.10.option.b
        goto = lc_orc_target
        
        ai_chance = { factor = 100 }
        
        trigger = {
            event_target:lc_orc_target = {
                NOT = { owned_by = ROOT }
            }
        }

        tooltip = {
            remove_local_concern = {
                TYPE = orc
            }
        }
        
		custom_tooltip = MIGRATION_ORCS_DIFFERENT_COUNTRY_TT
        
		hidden_effect = {
			event_target:lc_orc_target = {
				province_event = {
					id = wwu_lc_orcs.11
				}
			}
		}
    }
}
