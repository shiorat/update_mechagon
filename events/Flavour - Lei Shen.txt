# Resurrection of Lei Shen
country_event = {
	id = wwu_lei_shen.3
	title = wwu_lei_shen.3.title
	desc = wwu_lei_shen.3.desc
	picture = THRONE_OF_THUNDER_eventPicture
    
    is_triggered_only = yes
    
	trigger = {
        tag = P92 # Shan'ze
	}
	
	mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_lei_shen.3.option.a
        
        override_country_name = THRONE_OF_THUNDER
        
        define_ruler = {
            name = "Lei Shen"
            dynasty = "the Thunder King"
            min_age = 300
            max_age = 500
            adm = 5
            dip = 5
            mil = 7
        }
        
        add_country_modifier = {
            name = restoration_of_lei_shen
            duration = -1
        }
        
        continent_pandaria = {
            add_permanent_claim = ROOT
        }
        
        change_government = monarchy
        add_government_reform = autocracy_reform
        set_government_rank = 3
        
        hidden_effect = {
            set_ruler_flag = is_immortal
            
            setup_spellcaster_history = {
                APTITUDE = 4
                FINESSE = 4
                CONCENTRATION = 4
                MANA = 5
                shaman = yes
            }
        }
	}
}
