# N'Zoth's Influence
country_event = {
	id = wwu_naga.3
	title = wwu_naga.3.title
	desc = wwu_naga.3.desc
	picture = NZOTH_eventPicture
	
    is_triggered_only = yes
    
    trigger = {
        faction_in_power = faction_harbingers
        any_owned_province = {
            NOT = { religion = nzoth }
        }
    }

    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_naga.3.option.a
        
        random_owned_province = {
            limit = {
                NOT = { religion = nzoth }
            }
            change_religion = nzoth
        }
	}
}
