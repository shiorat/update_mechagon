# Elemental Rift
country_event = {
	id = wwu_invasion.1
	title = wwu_invasion.1.title
	desc = wwu_invasion.1.desc
    
	picture = {
		picture = RAGNAROS_eventPicture
        
		trigger = {
            OR = {
                culture_group = culture_group_fire
                religion = fire
            }
        }
	}
	picture = {
		picture = THERAZANE_eventPicture
        
		trigger = {
            OR = {
                culture_group = culture_group_earth
                religion = earth
            }
        }
	}
    picture = {
		picture = NEPTULON_eventPicture
        
		trigger = {
            OR = {
                culture_group = culture_group_water
                religion = water
            }
        }
	}
    picture = {
		picture = ALAKIR_THE_WINDLORD_eventPicture
        
		trigger = {
            OR = {
                culture_group = culture_group_air
                religion = wind
            }
        }
	}
    picture = {
        picture = BESTIARY_eventPicture
        
		trigger = {
            NOT = {
                OR = {
                    culture_group = culture_group_fire
                    religion = fire
                }
            }
            NOT = {
                OR = {
                    culture_group = culture_group_earth
                    religion = earth
                }
            }
            NOT = {
                OR = {
                    culture_group = culture_group_water
                    religion = water
                }
            }
            NOT = {
                OR = {
                    culture_group = culture_group_air
                    religion = wind
                }
            }
        }
	}
	
	trigger = {
        NOT = { has_country_flag = invasion_spawn_done }
        is_in_war = {
            casus_belli = cb_elemental_invasion
            attackers = ROOT
        }
        any_country = {
            war_with = ROOT
        }
    }
    
    immediate = {
        hidden_effect = {
            random_country = {
                limit = {
                    war_with = ROOT
                    is_in_war = {
                        casus_belli = cb_elemental_invasion
                        defenders = THIS
                    }
                }
                save_event_target_as = target_country
            }
        }
    }
    
    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_invasion.1.option.a
        
        custom_tooltip = FREE_TELEPORT_ARMY_TT
        hidden_effect = {
            set_country_flag = free_army_teleport
        }
	}
    
    after = {
        hidden_effect = {
            set_country_flag = invasion_spawn_done
        }
    }
}

# Fury of Ragnaros
country_event = {
	id = wwu_invasion.10
	title = wwu_invasion.10.title
	desc = wwu_invasion.10.desc
    picture = RAGNAROS_eventPicture
	
	trigger = {
        tag = P26
        is_in_war = {
            casus_belli = cb_elemental_invasion
            attackers = ROOT
        }
        any_country = {
            war_with = ROOT
            any_owned_province = {
                units_in_province = PREV
            }
        }
        NOT = { has_country_modifier = elemental_fatigue }
    }
    
    immediate = {
        hidden_effect = {
            random_country = {
                limit = {
                    war_with = ROOT
                    is_in_war = {
                        casus_belli = cb_elemental_invasion
                        defenders = THIS
                    }
                }
                save_event_target_as = target_country
            }
        }
    }
    
    mean_time_to_happen = {
		years = 25
	}
    
	option = {
		name = wwu_invasion.10.option.a
        
        event_target:target_country = {
            random_owned_province = {
                limit = {
                    units_in_province = event_target:target_country
                }
                kill_units = { 
                    who = owner
                }
            }
        }
        
        add_country_modifier = {
            name = elemental_fatigue
            duration = 3650
        }
	}
    
    option = {
		name = wwu_invasion.10.option.b
        
        add_mil_power = 25
	}
}

            tag = P28
            tag = P26
            tag = P24
            tag = P27
        }
        is_at_war = no
    }

    mean_time_to_happen = {
		years = 25
	}
    
	option = {
		name = wwu_invasion.100.option.a
        
        random_list = {
            # Easy Target
            50 = {
                random_country = {
                    limit = {
                        NOT = { total_development = 200 }

# AI - Invasion Helper - Teleport Army
country_event = {
	id = wwu_invasion.110
	title = wwu_invasion.110.title
	desc = wwu_invasion.110.desc
    picture = BESTIARY_eventPicture
	
    hidden = yes
    
	trigger = {
        ai = yes
        NOT = { has_country_modifier = ai_teleport_cooldown }
        
        OR = {  
            is_in_war = {
                casus_belli = cb_demonic_invasion
                attackers = ROOT
            }
            is_in_war = {
                casus_belli = cb_elemental_invasion
                attackers = ROOT
            }
        }
        
        any_owned_province = {
            num_of_units_in_province = {
                who = ROOT
                amount = 1
            }
        }
        
        any_country = {
            war_with = ROOT
            OR = {  
                is_in_war = {
                    casus_belli = cb_demonic_invasion
                    defenders = THIS
                }
                is_in_war = {
                    casus_belli = cb_elemental_invasion
                    defenders = THIS
                }
            }
            any_owned_province = {
                NOT = {
                    num_of_units_in_province = {
                        who = PREV
                        amount = 1
                    }
                }
            }
        }
    }

    mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            add_country_modifier = {
                name = ai_teleport_cooldown
                duration = 365
            }
            
            random_country = {
                limit = {
                    war_with = ROOT
                    OR = {  
                        is_in_war = {
                            casus_belli = cb_demonic_invasion
                            defenders = THIS
                        }
                        is_in_war = {
                            casus_belli = cb_elemental_invasion
                            defenders = THIS
                        }
                    }
                    any_owned_province = {
                        NOT = {
                            num_of_units_in_province = {
                                who = PREV
                                amount = 1
                            }
                        }
                    }
                }
                
                save_event_target_as = hostile_country
            }
        }
    }
    
	option = {
		name = wwu_invasion.110.option.a
        
        country_event = { id = wwu_teleport_system.10 }
	}
}
