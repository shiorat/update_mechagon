namespace = wwu_savage_vassalage

# Convert normal vassals into savage vassals for specific nations
country_event = {
	id = wwu_savage_vassalage.1
	title = wwu_savage_vassalage.1.title
	desc = wwu_savage_vassalage.1.desc
	picture = HORDE_MARCH_eventPicture

    hidden = yes

	trigger = { 
        NOT = { has_country_flag = ignore_tributary_vassal_event }
        is_subject = yes
        OR = {
            is_subject_of_type = vassal
            is_subject_of_type = march
        }
        overlord = {
            uses_tributary_vassalage = yes
        }
    }

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = wwu_savage_vassalage.1.option.a

        overlord = {
            create_subject = {
                subject_type = tributary_vassal
                subject = ROOT
            }
        }
	}
}
