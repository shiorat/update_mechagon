# Cycling of the Twin Emperor
country_event = {
	id = wwu_aqir.1
	title = wwu_aqir.1.title
	desc = wwu_aqir.1.desc
	picture = TWIN_EMPEROR_VEKLOR_eventPicture
    
	trigger = {
        OR = {
            tag = A97
            was_tag = A97
        }
		primary_culture = culture_qiraji
        NOT = { has_country_flag = has_secondary_emperor }
	}
	
	mean_time_to_happen = {
		months = 360
        
        modifier = {
            factor = 0.8
            
            NOT = { stability = 0 }
        }
	}
    
    immediate = {
        hidden_effect = {
            if = {
                limit = {
                    NOT = {
                        has_country_flag = setup_twin_emperor
                    }
                }
                set_country_flag = setup_twin_emperor
                define_exiled_ruler = {
                    exiled_as = secondary_emperor
                    name = "Vek'lor"
                    dynasty = "the Mighty"
                    min_age = 100
                }
            }
        }
    }
    
	option = {
		name = wwu_aqir.1.option.a
        
        hidden_effect = {
            presetup_transfer_spellcaster_history = yes
            exile_ruler_as = {
                name = primary_emperor
            }
        }
        
        set_ruler = secondary_emperor
        
        hidden_effect = {   
            set_country_flag = has_secondary_emperor
            transfer_spellcaster_history = yes
        }
        
        ai_chance = {
            factor = 100
        }
	}
    
    option = {
		name = wwu_aqir.1.option.b
        
		add_prestige = -20
        
        ai_chance = {
            factor = 0
        }
	}
}

# Hatred of the Titans
country_event = {
	id = wwu_aqir.3
	title = wwu_aqir.3.title
	desc = wwu_aqir.3.desc
	picture = RUINS_OF_AHN_QIRAJ_eventPicture
    
	trigger = {
		OR = {
            primary_culture = culture_qiraji
            primary_culture = culture_silithid
        }
        any_neighbor_country = {
            technology_group = tech_titan
        }
	}
	
	mean_time_to_happen = {
		months = 600
	}
    
    immediate = {
        hidden_effect = {
            random_neighbor_country = {
                limit = { 
                    technology_group = tech_titan 
                }
                
                save_event_target_as = target_country
            }
        }
    }

	option = {
		name = wwu_aqir.3.option.a
        
        declare_war_with_cb = {
            who = event_target:target_country
            casus_belli = cb_justified_war
        }
        
        ai_chance = {
            factor = 80
        }
	}
    
    option = {
		name = wwu_aqir.2.option.b
        
        add_stability = -1
        
        ai_chance = {
            factor = 20
        }
	}
}

# Crazed Lieutenant
country_event = {
	id = wwu_aqir.7
	title = wwu_aqir.7.title
	desc = wwu_aqir.7.desc
	picture = TENTACLES_FOR_ARMS_eventPicture
    goto = target_province
    
	trigger = {
		OR = {
            primary_culture = culture_qiraji
            primary_culture = culture_silithid
        }
        any_owned_province = {
            OR = {
                culture = culture_qiraji
                culture = culture_silithid
            }
        }
	}
	
	mean_time_to_happen = {
		months = 600
	}
    
    immediate = {
        hidden_effect = {
            random_owned_province = {
                limit = {
                    OR = {
                        culture = culture_qiraji
                        culture = culture_silithid
                    }
                }
                save_event_target_as = target_province
            }
        }
    }

	option = {
		name = wwu_aqir.7.option.a
        
        event_target:target_province = {
            spawn_rebels = {
                type = noble_rebels
                size = 3
                win = yes
            }
        }
	}
    option = {
		name = wwu_aqir.7.option.b
        
        add_mil_power = -100
        
        random = {
            chance = 25
            
            add_stability = -1
        }
	}
}
