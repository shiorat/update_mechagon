# Mining Plan
country_event = {
	id = wwu_cultural.8
	title = wwu_cultural.8.title
	desc = wwu_cultural.8.desc
	picture = FORGOTTEN_TORCH_eventPicture

	trigger = {
		advisor = natural_scientist
		any_owned_province = {
			trade_goods = gold
		}
        NOT = {
            capital_scope = {
                OR = {
                    continent = continent_outland
                    continent = continent_draenor
                }
            }
        }
	}
	
	mean_time_to_happen = {
		months = 1000
	}
	
	option = {
		name = wwu_cultural.8.opt1
		
		random_owned_province = {
			limit = { trade_goods = gold }
            
            if = {
                limit = {
                    NOT = { base_production = 12 }
                }
                add_base_production = 3
            }
            
            if = {
                limit = {
                    NOT = { base_tax = 12 }
                }
                add_base_tax = 1
            }
            
			change_culture = culture_kobold
		}
	}
	option = {
		name = wwu_cultural.8.opt2
	}
}
