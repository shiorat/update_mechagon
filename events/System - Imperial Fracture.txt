namespace = wwu_imperial_fracture

# The Imperial Fracture
country_event = {
	id = wwu_imperial_fracture.1
	title = wwu_imperial_fracture.1.title
	desc = wwu_imperial_fracture.1.desc
	picture = PANDAREN_SENTINEL_eventPicture
	
    major = yes
	is_triggered_only = yes
    
    trigger = {
        has_dlc = "Art of War"
        NOT = { has_global_flag = imperial_fracture_started }
        hre_leagues_enabled = no
		hre_religion_locked = no
		hre_religion_treaty = no
        is_emperor = yes
    }
	
    mean_time_to_happen = {
		days = 1
	}
    
    immediate = {
        hidden_effect = {
            set_global_flag = imperial_fracture_started
        }
    }
    
	option = {
		name = wwu_imperial_fracture.1.option.a
        
        enable_hre_leagues = yes
	}
}

# Restoration of the Empire (Celestials) 
country_event = {
	id = wwu_imperial_fracture.2
	title = wwu_imperial_fracture.2.title
	desc = wwu_imperial_fracture.2.desc
	picture = PANDAREN_WARRIOR_eventPicture
	
    major = yes
    
    trigger = {
        has_dlc = "Art of War"
        
        is_emperor = yes
		is_at_war = no
        religion = celestials
        
		NOT = {
			any_known_country = {
				is_in_league_war = yes
			}
		}		
        
		has_regency = no	
		hre_religion_locked = no
		hre_religion_treaty = no
        hre_leagues_enabled = yes
        
        had_global_flag = {
            flag = imperial_fracture_started
            days = 10950
        }
        
        OR = {
            AND = {
                NOT = {
                    any_known_country = {
                        is_elector = yes
                        OR = {
                            religion = sha
                        }
                        is_league_enemy = ROOT
                        truce_with = ROOT
                    }		
                }
            }
            had_global_flag = {
                flag = imperial_fracture_started
                days = 12775
            }
        }
    }
	
    mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_imperial_fracture.1.option.a
        
        set_hre_religion_locked = yes
	}
}

# Victory of the Sha
country_event = {
	id = wwu_imperial_fracture.20
	title = wwu_imperial_fracture.20.title
	desc = wwu_imperial_fracture.20.desc
	picture = SHA_ARISES_eventPicture
    
	major = yes

	is_triggered_only = yes
	
	trigger = {
		has_dlc = "Art of War"
		hre_religion = celestials
		OR = {
			NOT = { has_country_flag = hre_religion_changed }
			had_country_flag = { flag = hre_religion_changed days = 30 }
		}
	}
	
	immediate = {
		set_country_flag = hre_religion_changed
        
		hidden_effect = {
			set_hre_religion_treaty = yes
			set_hre_heretic_religion = celestials
			set_hre_religion = sha
			set_hre_religion_locked = yes
		}
	}

	option = {
		name = wwu_imperial_fracture.20.option.a
        
		tooltip = { 
			set_hre_heretic_religion = celestials
			set_hre_religion = sha	
			set_hre_religion_locked = yes			
		}

		set_hre_religion = sha
	}
}
