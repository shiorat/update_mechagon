# Minor Outbreak in [plague_ridden_province.GetName]
country_event = {
	id = wwu_pandemic.10
	title = wwu_pandemic.10.title
	desc = wwu_pandemic.10.desc
	picture = PLAGUE_eventPicture
    goto = plague_province
    
	is_triggered_only = yes
    
	trigger = {
		any_owned_province = {
            is_capital = no
        
            NOT = {
                has_province_modifier = plague_ridden_province
            }
        }
	}
    
    immediate = {
        hidden_effect = {
            random_owned_province = {
                limit = {
                    is_capital = no
                
                    NOT = {
                        has_province_modifier = plague_ridden_province
                    }
                }
                
                save_event_target_as = plague_province
            }
        }
    }
	
	mean_time_to_happen = {
		days = 1
	}
    
	option = {
		name = wwu_pandemic.10.option.a
		
		event_target:plague_province = {
            add_base_tax = -1
            
            if = {
                limit = {
                    NOT = { base_tax = 1 }
                }
                set_base_tax = 1
            }
            
            add_province_modifier = {
                name = plague_ridden_province
                duration = 1800
            }
        }
	}
    option = {
		name = wwu_pandemic.10.option.b
		
		event_target:plague_province = {
            add_base_production = -1
            
            if = {
                limit = {
                    NOT = { base_production = 1 }
                }
                set_base_production = 1
            }
            
            add_province_modifier = {
                name = plague_ridden_province
                duration = 1800
            }
        }
	}
    option = {
		name = wwu_pandemic.10.option.c
		
		event_target:plague_province = {
            add_base_manpower = -1
            
            if = {
                limit = {
                    NOT = { base_manpower = 1 }
                }
                set_base_manpower = 1
            }
            
            add_province_modifier = {
                name = plague_ridden_province
                duration = 1800
            }
        }
	}
    option = {
		name = wwu_pandemic.10.option.d
		
        add_yearly_manpower = -1
        
		event_target:plague_province = {
            add_province_modifier = {
                name = plague_ridden_province
                duration = 1800
            }
        }
	}
}

# Court Infection

	id = wwu_pandemic.14
	title = wwu_pandemic.14.title
	desc = wwu_pandemic.14.desc
	picture = PLAGUE_eventPicture
    
	is_triggered_only = yes
    
	trigger = {
        OR = {
            has_adm_advisor = yes
            has_dip_advisor = yes
            has_mil_advisor = yes
        }
	}

	mean_time_to_happen = {
		days = 1
	}
    
	
	option = {
		name = wwu_pandemic.14.option.a
		
		add_adm_power = -100
        
        random_list = {
            33 = {
                trigger = {
                    has_adm_advisor = yes
                }
                
                kill_advisor_by_category_effect = {
                    ADM = yes
                }
            }
            33 = {
                trigger = {
                    has_dip_advisor = yes
                }
                
                kill_advisor_by_category_effect = {
                    DIP = yes
                }
            }
            33 = {
                trigger = {
                    has_mil_advisor = yes
                }
                
                kill_advisor_by_category_effect = {
                    MIL = yes
                }
            }
            33 = {
                
            }
        }
	}
    option = {
		name = wwu_pandemic.13.option.b
		
		add_prestige = -25
        
        kill_advisor_by_category_effect = {
            ADM = yes
        }
        kill_advisor_by_category_effect = {
            DIP = yes
