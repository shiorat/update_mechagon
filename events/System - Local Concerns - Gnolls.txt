# Local Migration
province_event = {
    id = wwu_lc_gnolls.10
    title = wwu_lc_gnolls.10.title
    desc = wwu_lc_gnolls.10.desc
    picture = RIVERPAW_GNOLL_eventPicture
    goto = ROOT
    
    is_triggered_only = yes
    
    trigger = {
        OR = {
            has_local_concern_level_1 = { TYPE = gnoll }
            has_local_concern_level_2 = { TYPE = gnoll }
        }

        any_neighbor_province = {
            is_city = yes
            
            has_no_local_concern = {
                TYPE = gnoll
            }
            
            NOT = {
                owner = {
                    is_gnoll_type_nation = yes
                }
            }

            NOT = {
                has_banned_terrain = yes
            }
        }
    }
    
    immediate = {
        hidden_effect = {
            random_neighbor_province = {
                limit = {
                    is_city = yes

                    has_no_local_concern = {
                        TYPE = gnoll
                    }
                    
                    NOT = {
                        owner = {
                            is_gnoll_type_nation = yes
                        }
                    }
                    
                    NOT = {
                        has_banned_terrain = yes
                    }
                }
                
                save_event_target_as = lc_gnoll_target
            }
        }
    }

    # Same Country
    option = {  
        name = wwu_lc_gnolls.10.option.a
        goto = lc_gnoll_target
        
        ai_chance = { factor = 100 }
        
        trigger = {
            event_target:lc_gnoll_target = {
                owned_by = ROOT
            }
        }

        # Remove gnolls
        tooltip = {
            remove_local_concern = {
                TYPE = gnoll
            }
        }

        # Add gnolls to the new province based on the previous province tier
        event_target:lc_gnoll_target = {
            if = {
                limit = {
                    owned_by = ROOT
                }

                upgrade_local_concern = {
                    TYPE = gnoll
                }
                
                # Remove gnolls from the previous province
                PREV = {
                    remove_local_concern = {
                        TYPE = gnoll
                    }
                }
            }
        }
    }

    # Different Country
    option = {  
        name = wwu_lc_gnolls.10.option.b
        goto = lc_gnoll_target
        
        ai_chance = { factor = 100 }
        
        trigger = {
            event_target:lc_gnoll_target = {
                NOT = { owned_by = ROOT }
            }
        }

        tooltip = {
            remove_local_concern = {
                TYPE = gnoll
            }
        }
        
		custom_tooltip = MIGRATION_GNOLLS_DIFFERENT_COUNTRY_TT
        
		hidden_effect = {
			event_target:lc_gnoll_target = {
				province_event = {
					id = wwu_lc_gnolls.11
				}
			}
		}
    }
}
