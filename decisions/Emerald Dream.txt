country_decisions = {
	awaken_from_dream = {
		potential = {
			has_country_modifier = dreaming
			religion = druidism
		}
		allow = {
			stability = 1
		}
		effect = {
			add_stability = -1
            
			remove_country_modifier = dreaming
            
            if = {
                limit = {
                    primary_culture = culture_green_dragonflight
                }
                add_country_modifier = {
					name = slumber_sickness
					duration = 3650
				}
            }
            else = {
                add_country_modifier = {
					name = slumber_sickness
					duration = 7300
				}
            }
		}
		ai_will_do = {
			factor = 1
            
            # Don't awaken if at peace
			modifier = {
				factor = 0
                
				NOT = { is_at_war = yes }
			}
            
            # Don't awaken if not at full Dream
            modifier = {
				factor = 0
                
				NOT = { piety = 1.0 }
			}
		}
	}
	
	slumber_into_dream = {
		potential = {
			NOT = { has_country_modifier = dreaming }
			religion = druidism
		}
		allow = {
            stability = 0
			NOT = { is_at_war = yes }
		}
		effect = {
			custom_tooltip = slumber_into_dream_tooltip
            
			add_country_modifier = {
				name = dreaming
				duration = -1
			}
		}
		ai_will_do = {
			factor = 1
            
            # Don't use if Purity isn't very low
			modifier = {
				factor = 0
                
				piety = -0.25
			}
		}
	}
}
